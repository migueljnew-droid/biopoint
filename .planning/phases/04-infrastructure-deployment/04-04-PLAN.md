**Step 1: Create `fly.toml` at `/Users/GRAMMY/biopoint/apps/api/fly.toml`**

This file configures the biopoint-api Fly.io app. The existing Dockerfile is at `apps/api/Dockerfile` and is already built for `node:20-alpine` — correct for Fly.io.

```toml
# Fly.io app configuration for biopoint-api
# Deploy from apps/api/ directory: fly deploy --config fly.toml
app = 'biopoint-api'
primary_region = 'iad'  # us-east-1 — closest to Neon default region

[build]
  dockerfile = 'Dockerfile'

[env]
  PORT = '3000'
  NODE_ENV = 'production'

[http_service]
  internal_port = 3000
  force_https = true
  auto_stop_machines = false   # HIPAA: no cold starts; PHI must be available without latency spikes
  auto_start_machines = true
  min_machines_running = 1     # Keep one instance always running

[[http_service.ports]]
  port = 443
  handlers = ['tls', 'http']

[[http_service.ports]]
  port = 80
  handlers = ['http']

[[vm]]
  memory = '512mb'
  cpu_kind = 'shared'
  cpus = 1

[checks]
  [checks.health]
    grace_period = '10s'
    interval = '30s'
    method = 'GET'
    path = '/health'
    port = 3000
    timeout = '5s'
    type = 'http'
```

**Step 2: Create `redis-fly.toml` at `/Users/GRAMMY/biopoint/redis-fly.toml`**

This file deploys self-hosted Redis as a separate Fly.io app within the same organization. Using self-hosted Redis ensures Fly.io's BAA covers the data (vs. Upstash which requires its own Enterprise BAA for HIPAA).

```toml
# Fly.io configuration for self-hosted Redis
# Used for HIPAA-compliant rate limiting state persistence
# Deploy separately: fly deploy --config redis-fly.toml --app biopoint-redis
#
# IMPORTANT: Deploy in the SAME Fly.io organization as biopoint-api
# so that the private network (biopoint-redis.internal) is accessible.
#
# Setup commands:
#   fly launch --name biopoint-redis --org biopoint-production --image redis:7-alpine --no-deploy
#   fly volume create redis_data --region iad --size 1 --app biopoint-redis
#   fly deploy --config redis-fly.toml
app = 'biopoint-redis'
primary_region = 'iad'

[build]
  image = 'redis:7-alpine'

[[mounts]]
  source = 'redis_data'
  destination = '/data'

# NO [http_service] — Redis is private network only.
# biopoint-api connects via: biopoint-redis.internal:6379

[[vm]]
  memory = '256mb'
  cpu_kind = 'shared'
  cpus = 1
```

**To deploy to production, execute these steps in order:**

---

### Step 1: Install flyctl (if not installed)
```bash
brew install flyctl
fly auth login
```

### Step 2: Set up Fly.io organization and sign BAA
1. Create a production-isolated organization:
   ```bash
   fly orgs create biopoint-production
   ```
2. Upgrade to paid plan (required for BAA): **fly.io/pricing**
3. Sign HIPAA BAA ($99/month): **fly.io/dashboard/biopoint-production/compliance**
4. Select "HIPAA Compliance Package" → Sign BAA → Confirm

### Step 3: Enable Neon HIPAA on your database project
**WARNING: This operation is IRREVERSIBLE. Do this only on your production Neon project.**
1. Upgrade to Neon Scale plan: **neon.com/pricing**
2. In Neon console: **Project Settings → HIPAA Support → Accept BAA**
3. Note the connection strings (both pooled and direct URLs)

### Step 4: Deploy self-hosted Redis
```bash
cd /Users/GRAMMY/biopoint

# Create Redis app in production org
fly launch --name biopoint-redis --org biopoint-production --image redis:7-alpine --no-deploy

# Create persistent volume
fly volume create redis_data --region iad --size 1 --app biopoint-redis

# Deploy Redis
fly deploy --config redis-fly.toml
```

### Step 5: Create and configure biopoint-api app
```bash
cd /Users/GRAMMY/biopoint/apps/api

# Create the API app
fly launch --name biopoint-api --org biopoint-production --no-deploy

# Set all production secrets (replace values with your actual secrets):
fly secrets set \
  DATABASE_URL="postgres://user:pass@ep-xxx-pooler.us-east-2.aws.neon.tech/neondb?sslmode=require" \
  DIRECT_URL="postgres://user:pass@ep-xxx.us-east-2.aws.neon.tech/neondb?sslmode=require" \
  JWT_SECRET="your-production-jwt-secret-min-32-chars" \
  JWT_ACCESS_EXPIRES="15m" \
  JWT_REFRESH_EXPIRES="7d" \
  REDIS_HOST="biopoint-redis.internal" \
  REDIS_PORT="6379" \
  NODE_ENV="production" \
  PORT="3000" \
  AWS_REGION="auto" \
  AWS_ACCESS_KEY_ID="your-r2-key" \
  AWS_SECRET_ACCESS_KEY="your-r2-secret" \
  S3_BUCKET="biopoint-uploads-prod" \
  S3_ENDPOINT="https://account-id.r2.cloudflarestorage.com" \
  CORS_ORIGIN="https://app.biopoint.health" \
  SENTRY_DSN="your-sentry-dsn" \
  SENTRY_ENABLED="true" \
  DATADOG_API_KEY="your-datadog-api-key" \
  DATADOG_ENABLED="true" \
  SERVICE_VERSION="1.0.0" \
  HEALTH_CHECK_TOKEN="your-health-check-token" \
  --app biopoint-api
```

### Step 6: Run database migrations
```bash
# From the db directory, using DIRECT_URL for migration (not pooled URL)
cd /Users/GRAMMY/biopoint/db
DIRECT_URL="your-direct-neon-url" npx prisma migrate deploy
```

### Step 7: Deploy the API
```bash
cd /Users/GRAMMY/biopoint/apps/api
fly deploy --app biopoint-api
```
